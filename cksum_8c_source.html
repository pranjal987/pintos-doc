63 cksum (const void *b_, size_t n)
64 {
65  const unsigned char *b = b_;

68  for (i = n; i > 0; --i)
69  {
70  unsigned char c = *b++;
71  s = (s << 8) ^ crctab[(s >> 24) ^ c];
72  }
73  while (n != 0)
74  {
75  unsigned char c = n;

77  s = (s << 8) ^ crctab[(s >> 24) ^ c];

80 }
